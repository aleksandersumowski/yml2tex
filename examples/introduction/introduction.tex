\section{Introduction}
\subsection{Why?}
\frame {
	\frametitle{Why?}
	\begin{itemize}[<+-| alert@+>]
	\item Creating Presentations in PowerPoint/Keynote etc. takes too long
	\begin{itemize}[<+-| alert@+>]
	\item No automatic TOC generation
	\item No Code Highlighting
	\item Applications not for free
	\item Proprietary formats
	\end{itemize}
	\end{itemize}
}

\section{Features}
\subsection{General}
\frame {
	\frametitle{General}
	\begin{itemize}[<+-| alert@+>]
	\item Easy \& fast to generate simple presentations
	\item Adds a nice default style to the document
	\item Generates a TOC out of sections/subsections
	\item Separates content from presentation
	\end{itemize}
}
